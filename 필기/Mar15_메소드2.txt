f(n) = {
    1 (n == 1)
    n + f(n - 1) (n > 1)
}
재귀형 프로그래밍.

public static int sum(int n){
    if (n == 1)
        return 1;

    return n + sum(n - 1);
}

public static int fibonacci(int n){
    if (n == 1 || n == 2)
        return 1;

    return fibonacci(n - 1) + fibonacci(n - 2);
}
